Code Fencing

Dijkstra's algorithm is a greedy algorithm for the shortest path from a single vertex. It grows a spanning tree in the graph always adding the edge that connects the start vertex with a minimal path length to the spanning tree. Dijkstra's algorithm uses a priority queue of vertices key by the distance estimation. The distances from the start vertex are update by a relaxation process
Dijkstra's Algorithm 
const int INF = 1000000000;
    vector>> adj;
    
    void dijkstra(int s, vector & d, vector & p) {
        int n = adj.size();
        d.assign(n, INF);
        p.assign(n, -1);
    
        d[s] = 0;
        set> q;
        q.insert({0, s});
        while (!q.empty()) {
            int v = q.begin()->second;
            q.erase(q.begin());
    
            for (auto edge : adj[v]) {
                int to = edge.first;
                int len = edge.second;
    
                if (d[v] + len < d[to]) {
                    q.erase({d[to], to});
                    d[to] = d[v] + len;
                    p[to] = v;
                    q.insert({d[to], to});
                }
            }
        }
    }
